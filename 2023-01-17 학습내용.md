## 🔎 1. 할인 가격 프로그램
```html,js
<body>
    <div id="contents">
        <img src="#">
        <!--이미지를 직접 넣어보고 시작해도 괜찮다! -->
            <ul>
                <li>
                    <label for="originalPrice">원래가격</label>
                    <input type="text" id="originalPrice">원
                </li>

                <li>
                    <label for="salePrice">할인율</label>
                    <input type="text" id="salePrice">%
                </li>

                <li>
                    <button onclick="showPrice()">할인가격 계산하기</button>
                </li>
            </ul>
    </div>

    <div id="showResult"></div>

<script>
    function showPrice(){
        var oPrice = document.querySelector("#originalPrice").value; //웹 요소를 먼저 가져옴
        var sale = document.querySelector("#salePrice").value; //웹 요소를 먼저 가져옴

        if(oPrice > 0 && sale > 0) //둘 다 양수라면?
        {
            var savedPrice = oPrice * (sale/100);
            var resultPrice = oPrice - savedPrice;
        }
        document.querySelector("#showResult").innerHTML = "상품의 원래 가격은 "+ oPrice + "이며 할인된 가격은 " + resultPrice + "입니다.";
    }
</script>    

```
> 주석처리로 설명을 정리해놨으니 참고하면 된다.
