## 🔎 1. 실습 - if문, if...else문, 조건 연산자 예제
```html,js

 // 3의 배수 프로그램을 작성하여라
<div id="contents"></div>

    <script>
        var num = prompt("숫자를 입력해주세요");
        var numArea = document.querySelector("#contents");

        if(num%3==0){
            numArea.innerHTML = "3의 배수입니다"; //innerHTML을 이용하여 HTML코드로 출력한다.
        }
        else numArea.innerHTML = "3의 배수가 아닙니다."; //innerHTML을 이용하여 HTML코드로 출력한다.
    </script>
```

### 1-1 짝수 검사하는 프로그램
```html,js
<script>
        var value = window.prompt("숫자를 입력해주세요!");

        value = parseInt(value);
        //prompt()로 받게되면 string형으로 받기 때문에 반드시 형변환을 해줘야한다.
        //alert(typeof(value))
        //prompt()에서 받은 value값은 string타입이고 parseInt() 통해서 
        //string타입을 강제로 정수타입으로 바꾸고 있다(강제형변환) 
        //정수로 형변환할 때 사용하는 건 parseInt();
 
        if(value % 2 == 0){
            alert("짝수입니다."); //알림창에서 결과를 표출
        }
        else{
            alert("홀수입니다."); //알림창에서 결과를 표출
        }
    </script>
```
