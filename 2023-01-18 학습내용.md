### 1-2 랜덤 난수 발생기 프로그램
```html,js
<script>
var low = 0; //범위의 최소값
        var high = 0; // 범위의 최고값
        var card = 0; //정답
        

        while(true){
            var count = 0; //사용자의 시도 횟수
            
            low = 0;
            high = 99;
            card = Math.floor(Math.random() * 100);  // 0~99까지의 난수를 발생시킨다.
            //random()함수는 1미만으로 랜덤으로 소수로 난수가 생긴다
            //그래서 100을 곱해주는거다!
            //floor()함수는 밑에 소수점을 다 날려버리는거다!

            alert("수를 결정하였습니다. 맞춰보세요.");
            
            while(true){
                var n = 0;
                alert(low + " - "+ high);
                count++;
                alert("시도횟수: "+ count + "번째");

                n = prompt("숫자를 입력해주세요!"); //사용자로부터 숫자를 입력받음
                n = parseInt(n); //강제 형변환

                if(n > high || n < low){
                    alert("값의 범위를 벗어났어요.");
                }
                else{
                    if(n == card){
                        alert("정답입니다.짝짝짝!");
                        break;
                    }

                    else if(n>card){
                        alert("정답은 더 낮은 수 입니다.");
                        high = n; //범위를 재지정 해줍니다.
                    }

                    else {
                        alert("정답은 더 높은 수 입니다.");
                        low = n;
                    }
                }
            }
            alert("시도횟수는 "+ count);

            var con = confirm("다시 게임을 시작하겠습니까?");
            //무한 루프를 사용할 경우에는 반드시 빠져나갈 곳을 만들것!
            if(con == false){
                break;
            }
        }
 </script>
```

### 1-3 do-while 실질적 사용예제
```html,js
<script>
        var inputString = null;

        alert("메시지를 입력해주세요");
        alert("프로그램을 종료하려면 q를 입력해주세요");
        //do-while문은 무조건 실행된다.
        do{
            inputString = prompt("메시지를 입력하세요.");
            document.write(inputString + "<br/>");
        } while(!(inputString== "q")) 
        /* q를 입력하면 true가 되어 루프를 계속 돌게 된다. */
        /* 그래서 논리부정연산자를 이용하여 q를 입력하면 false로 되어 탈출하도록 진행하였다.*/

        document.write("프로그램 종료!!");
    </script>    

```

### 1-5 continue 실질적 사용예제2
```html,js
 <script>
        //2와 3의 공배수를 제외하고 다 출력하게끔 하는 코드
        for(var i=1; i<=100; i++){
            //2와 3의 공배수인지 확인하는 코드
            if(i%2==0 && i%3==0){
                continue;
                //continue를 실행하게 되면 continue아래에 있는 실행코드를
                //무시하고 증감식으로 이동한다.
            }
            document.write(i + "<br/>");
            //조건문에서 걸러진 애들을 출력한다.
        }

    </script>

```
