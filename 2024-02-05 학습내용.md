### 실습-2 (함수 선언의 종류)
```js

//선언식
  function sum() {
    let result = 10 + 10;
  }

  // 밑 표현식처럼 sum이 선언되지 않은 상태에서 호출을 하면 당연히 호출이 안되지만
  // 선언식에서는 무조건 function이 맨위에 있는 것처럼 바로 실행이 된다.(코드만 있다면 실행이 된다는 것이다)

//표현식
  sum()
  const sum = function() {
    let result = 10 + 10;
  }
  //선언하기전에 실행을하면 오류가 남
  //선언문을 하게되면 예측어려운 동작이 진행되는데 표현식은 그렇지 않는다.
  
  
//화살표 함수
 const sum = () => { //function 생략
  let result = 10 + 10;
 }
 //메소드 내에서 함수를 사용하는데에 많이 사용된다.
 //표현식과 같이 돌아가게 된다.
 
  
```
